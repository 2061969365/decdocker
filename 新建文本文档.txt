1. .github/workflows/docker-publish.yml
YAML
name: 🚀 自动构建 Docker 镜像并发布到 GHCR

on:
  push:
    branches:
      - main
  workflow_dispatch:

jobs:
  build-and-push:
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write

    steps:
      - name: 📂 Check out the repo
        uses: actions/checkout@v4

      - name: 🛠️ Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: 🔑 Log in to GitHub Container Registry
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: 🔤 Downcase repository name
        run: |
          LOW_NAME=$(echo "ghcr.io/${{ github.repository }}" | tr '[:upper:]' '[:lower:]')
          echo "IMAGE_NAME=${LOW_NAME}" >> $GITHUB_ENV

      - name: 📦 Build and push Docker image
        uses: docker/build-push-action@v5
        with:
          context: .
          file: ./Dockerfile
          push: true
          tags: |
            ${{ env.IMAGE_NAME }}:latest
          cache-from: type=gha
          cache-to: type=gha,mode=max

FROM teddysun/xray:latest AS xray-source
FROM cloudflare/cloudflared:latest AS cf-source

# 最终运行的极简 Alpine 生产环境
FROM alpine:latest
RUN apk add --no-cache bash curl busybox-extras

WORKDIR /app

# 灌入跨平台核心可执行程序
COPY --from=xray-source /usr/bin/xray /usr/bin/xray
COPY --from=cf-source /usr/local/bin/cloudflared /usr/local/bin/cloudflared

COPY . .

# 强行清洗 Windows 换行符隐形毒素并赋予执行权
RUN sed -i 's/\r$//' /app/start.sh
RUN chmod +x /app/start.sh

ENTRYPOINT ["/app/start.sh"]
3. start.sh
Bash
#!/bin/bash

# 1. 强制锁死核心通用密钥 UUID
export UUID="a29738e5-bee1-c0fc-b484-ae7c49cbc828"
echo "🔑 核心 UUID 密码已强制锁定为固定值: $UUID"

# ================= 动态公网 IP 与位置抓取组件 =================
echo "🔍 正在打捞当前容器的真实外网出口 IP 与归属地..."
REAL_IP=$(curl -s --max-time 3 ifconfig.me)
REAL_COUNTRY=$(curl -s --max-time 3 ipinfo.io/country)

if [ -z "$REAL_IP" ]; then REAL_IP="DynamicIP"; fi
if [ -z "$REAL_COUNTRY" ]; then REAL_COUNTRY="Cloud"; fi

# 组合出专属于当前主机机房的即时别名后缀
NODE_REMARK="${REAL_COUNTRY}_${REAL_IP}"
echo "📍 探测成功！当前主机实际地理标记: $NODE_REMARK"
# =====================================================================

# 动态将固定 UUID 和动态节点名字注入核心配置和网页
sed -i "s/UUID_PLACEHOLDER/$UUID/g" /app/config.json
sed -i "s/UUID_PLACEHOLDER/$UUID/g" /app/www/index.html
sed -i "s/NODE_REMARK_PLACEHOLDER/$NODE_REMARK/g" /app/www/index.html

# 2. 启动内部增强网页服务器（监听内网 8081，不对公网开洞）
httpd -p 8081 -h /app/www &
echo "🌐 静态网页服务已在内部 8081 端口挂载"

# 3. 启动 Xray 核心组件（总闸口 8080）
/usr/bin/xray -config /app/config.json &
echo "🚀 Xray 核心网关已在本地 8080 端口拉起"

# ================= 🎛️ 三模自适应隧道启动网关 =================
echo "🚇 正在解析云端隧道环境变量..."

if [ -n "$TUNNEL_TOKEN" ]; then
  # 玩法一：标准的平台全局 Token 模式
  echo "👉 模式 [A] 激活：检测到 TUNNEL_TOKEN，正在建立官方固定隧道..."
  /usr/local/bin/cloudflared tunnel --edge-ip-version auto --no-autoupdate --protocol http2 run --token "$TUNNEL_TOKEN" &

elif [ -n "$ARGO_AUTH" ] && [ -n "$ARGO_DOMAIN" ]; then
  # 玩法二：高级自定义全配模式（完美兼容 Json 证书或独立 Token）
  if [[ "$ARGO_AUTH" =~ TunnelSecret ]]; then
    echo "👉 模式 [B-1] 激活：检测到 JSON 证书凭证，正在本地流式重组隧道..."
    echo "$ARGO_AUTH" > /app/tunnel.json
    
    # 从 JSON 证书中动态提取 TunnelID 并拼装临时 yaml 配置文件
    TUNNEL_ID=$(echo "$ARGO_AUTH" | grep -oE '"TunnelID":"[^"]+"' | cut -d'"' -f4)
    cat <<EOF > /app/tunnel.yml
tunnel: $TUNNEL_ID
credentials-file: /app/tunnel.json
protocol: http2
ingress:
  - hostname: $ARGO_DOMAIN
    service: http://localhost:8080
  - service: http_status:404
EOF
    /usr/local/bin/cloudflared tunnel --config /app/tunnel.yml run &
  else
    echo "👉 模式 [B-2] 激活：检测到 Token 形式的 ARGO_AUTH，正在向域名 $ARGO_DOMAIN 绑定大桥..."
    /usr/local/bin/cloudflared tunnel --edge-ip-version auto --no-autoupdate --protocol http2 run --token "$ARGO_AUTH" &
  fi

else
  # 玩法三：纯粹的零门槛白嫖 TryCloudflare 临时快速隧道模式
  echo "👉 模式 [C] 激活：未检测到任何固定密钥，正在拉起 TryCloudflare 临时随机隧道..."
  /usr/local/bin/cloudflared tunnel --edge-ip-version auto --no-autoupdate --protocol http2 --url http://localhost:8080 &
fi
# =====================================================================

# 4. 极致精简版总闸雷达监控循环（拦截死锁，防范算术污染报错）
while true; do
  sleep 15
  
  # 只需要盯着 8080 入口总闸和隧道守护进程是否活着即可
  netstat -tln | grep -q :8080
  XRAY_GATE=$?
  pidof cloudflared > /dev/null
  CF_PROCESS=$?
  
  if [ $XRAY_GATE -ne 0 ] || [ $CF_PROCESS -ne 0 ]; then
    echo "🚨 【断流警报】检测到服务硬断流！(总入口:$XRAY_GATE, 隧道进程:$CF_PROCESS)"
    exit 1
  fi
done

        function generateLinks(connectAddress) {
            const vlessLink = `vless://${uuid}@${connectAddress}:443?encryption=none&security=tls&type=ws&host=${currentHost}&sni=${currentHost}&path=${encodeURIComponent(wsPath)}#${nodeRemark}`;
            document.getElementById('nodeLink').innerText = vlessLink;
            document.getElementById('subContent').innerText = btoa(unescape(encodeURIComponent(vlessLink)));
        }

        function applyCustomIp() {
            const inputVal = document.getElementById('cfIpInput').value.trim();
            generateLinks(inputVal ? inputVal : currentHost);
            alert(inputVal ? `已成功套用接入点: ${inputVal}` : '已恢复默认隧道域名解析');
        }

        function fetchOptimizedIPs() {
            fetch('https://cdn.jsdelivr.net/gh/Alvin9999/new-pac@master/cloudflarespeedtest/cloudflare.txt')
                .then(res => res.text())
                .then(text => {
                    const lines = text.split('\n').map(s => s.trim()).filter(s => s.length > 0 && !s.includes(':'));
                    const poolDiv = document.getElementById('ipPool');
                    poolDiv.innerHTML = ''; 
                    
                    lines.slice(0, 6).forEach(ip => {
                        const span = document.createElement('span');
                        span.className = 'ip-tag';
                        span.innerText = ip;
                        span.onclick = () => {
                            document.getElementById('cfIpInput').value = ip;
                            generateLinks(ip);
                            alert(`✨ 已成功切换到该优选 IP: ${ip}`);
                        };
                        poolDiv.appendChild(span);
                    });
                }).catch(() => {
                    document.getElementById('ipPool').innerHTML = '<span style="font-size:12px; color:#ef4444;">公网资源池打捞失败。</span>';
                });
        }

        function copyText(elementId) {
            const text = document.getElementById(elementId).innerText;
            navigator.clipboard.writeText(text).then(() => alert('已成功复制到剪贴板！'));
        }
